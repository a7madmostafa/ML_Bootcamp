def cumul_cases(country, start_date, end_date):

    fig=px.line(df[(df.country==country) & (df.date_reported>=start_date) & (df.date_reported<=end_date)],
                x="date_reported", y="cumulative_cases",
                title=f"cumulative cases over time for {country} from {start_date} to {end_date}")
    fig.show()
